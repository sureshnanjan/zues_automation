 function palindrome(number){ 
   var number,rev,rem,original;   
   original=number;  
 while(number>0)
 {
  rem=number%10;
  rev=(rev*10)+rem;
  number=number/10;
 }
 if(original==rev)
 return flag=true;
 else return flag=false;
}
